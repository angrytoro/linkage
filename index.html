          <div class="form-group">
            <label class="radio-inline">
              <input type="radio" name="demon1" value="1" checked="checked">开启
            </label>
            <label class="radio-inline">
              <input type="radio" name="demon1" value="2" >关闭
            </label>
          </div>
          <div class="form-group">
            <label class="checkbox-inline">
              <input type="checkbox" name="demon2" value="1" checked="checked">value=1
            </label>
            <label class="checkbox-inline">
              <input type="checkbox" name="demon2" value="2">value=2
            </label>
            <label class="checkbox-inline">
              <input type="checkbox" name="demon2" value="3">value=3
            </label>
          </div>
          <div class="form-group" data-matchable='m1[1]&m2[1]' data-matchable-type="disabled">
            <label for="">当m1=1，m2=1的时候可以编辑，否则是disabled</label>
            <input type="text" class="form-control">
          </div>
          <div class="form-group" data-matchable="m1[2]|m2[2]" data-matchable-type="active">
            <label for="">当m1=2或者m2=2的时候显示，否则是隐藏的，并且是disabled的</label>
            <input type="text" class="form-control">
          </div>
          <div class="form-group" data-matchable="m2[3]">
            <label for="">默认不设置data-matchable-type的时候，相应的表单元素是不会disabled的。</label>
            <input type="text" class="form-control">
          </div>
            
            
              $.matchable.register('m1',$('[name="demon1"]'), 'radio');
              $.matchable.register('m2',$('[name="demon2"]'), 'checkbox');
